#include <iostream>
using namespace std;

int main() {
    int i, b;
    bool isPrime = true;

    cout << "Enter a positive integer: ";
    cin >> b;

    // 0 and 1 are not prime numbers
    if (b == 0 || b == 1) {
        isPrime = false;
    }
    else {
        for (i = 2; i <= b / 2; ++i) {
            if (b % i == 0) {
                isPrime = false;
                break;
            }
        }
    }
    if (isPrime)
        cout << b << " is a prime number";
    else
        cout << b << " is not a prime number";

    return 0;
}
